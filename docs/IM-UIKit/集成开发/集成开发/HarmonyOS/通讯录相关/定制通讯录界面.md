网易云信 IM UIKit 的通讯录模块（[`contactkit-ui`](https://github.com/netease-kit/nim-uikit-harmony/tree/main/contactkit_ui)）包含好友列表、验证消息、黑名单、我的群聊等界面（UI）。本文以好友列表为例，介绍如何定制界面和交互事件单击逻辑。

## 组件说明

在通讯录页面（`ContactPage`）中，按照 UI 样式划分为 **功能模块组件** 和 **好友列表组件** 两种类别。

## 实现方式

在通讯录组件中，包含两个 `List`，分别是功能模块和好友列表。其中好友列表由多个 `ListItemGroup` 分组构成，每个分组下包含同一首字母的好友列表，可参考如下示例代码：

```TypeScript
  ...
  // 功能模块
  List() {
    ForEach(this.headerItems, (item: NEContactItem) => {
      ListItem() {
        ContactItemBuilder(item)
      }
    })
  }

  // 好友列表
  List({ scroller: this.listScroller }) {
    ForEach(this.viewModel.friendSection, (item: NEContactSection, index) => {
      // 昵称首字母分组
      ListItemGroup({
        header: ListGroupHeader(item.initial)
      }) {
        // 好友列表
        LazyForEach(item.contacts, (item: NEUserWithFriend, index) => {
          ListItem() {
            ViewItemBuilder({
              ...
            })
          }
        }, (item: NEUserWithFriend, index) => item.user?.accountId)
      }
    })
  }
  ...
```

## 定制通讯录组件界面

您可以通过修改源码，实现定制 UI 组件，达到预期效果。例如：

- 适用场景：修改通讯录组件中的头像大小、好友名称的字体大小。

- 实现方式：修改代码位于 [src/main/ets/pages/ContactPage.ets](https://github.com/netease-kit/nim-uikit-harmony/blob/main/contactkit_ui/src/main/ets/pages/ContactPage.ets) `build` 方法中的 `ViewItemBuilder` 入参，将 `titleIconSize` 参数改为预期值即可。

- 示例代码：

    ```TypeScript
    ...
    // 好友列表
    List({ scroller: this.listScroller }) {
        ForEach(this.viewModel.friendSection, (item: NEContactSection, index) => {
        // 昵称首字母分组
        ListItemGroup({
            header: ListGroupHeader(item.initial)
        }) {
            // 好友列表
            LazyForEach(item.contacts, (item: NEUserWithFriend, index) => {
            ListItem() {
                ViewItemBuilder({
                ...
                titleIconSize: 36, // 头像大小
                titleFontSize: 14, // 好友名称的字体大小
                ...
                })
            }
            }, (item: NEUserWithFriend, index) => item.user?.accountId)
        }
        })
    }
    ...
    ```

## 定制通讯录组件事件

通讯录单击事件的实现逻辑，可参考通讯录页面 `ContactPage` 中的具体实现。在通讯录中的单击事件的实现示例代码如下所示：

```TypeScript
  ...
  // 功能模块
  List() {
    ForEach(this.headerItems, (item: NEContactItem) => {
      ListItem() {
        ContactItemBuilder(item)
      }
      ...
      .onClick(() => { // 功能模块单击事件
        this.pathStack.pushPath({
          name: item.routerUrl,
          onPop: (popInfo: PopInfo) => {
            this.loadUnreadApplication()
          }
        })
      })
      ...
    })
  }

  // 好友列表
  List({ scroller: this.listScroller }) {
    ForEach(this.viewModel.friendSection, (item: NEContactSection, index) => {
      // 昵称首字母分组
      ListItemGroup({
        header: ListGroupHeader(item.initial)
      }) {
        // 好友列表
        LazyForEach(item.contacts, (item: NEUserWithFriend, index) => {
          ListItem() {
            ViewItemBuilder({
              ...
            })
          }
          ...
          .onClick(() => { // 好友列表单击事件
            this.pathStack.pushPath({
              name: 'PersonInfoPage',
              param: item.user?.accountId
            })
          })
          ...
        }, (item: NEUserWithFriend, index) => item.user?.accountId)
      }
    })
  }
  ...
```
