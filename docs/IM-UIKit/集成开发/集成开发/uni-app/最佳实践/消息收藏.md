## 开启消息收藏功能

在 UIKit 初始化时，将 `localOptions` 中的 `loginStateVisible` 配置为 true。

```
//初始化uikit
const nim = (uni.$UIKitNIM = V2NIM.getInstance(
{
    appkey: "your app key",
    needReconnect: true,
    debugLevel: "debug",
    apiVersion: "v2",
},
{
    V2NIMLoginServiceConfig: {
    lbsUrls: isWxApp
        ? ["https://lbs.netease.im/lbs/wxwebconf.jsp"]
        : ["https://lbs.netease.im/lbs/webconf.jsp"],
    linkUrl: isWxApp ? "wlnimsc0.netease.im" : "weblink.netease.im",
    isFixedDeviceId: true,
    },
}
));

const store = (uni.$UIKitStore = new RootStore(
nim,
{
    // 启用用户在线状态
    loginStateVisible: true
},
"UniApp"
));

//登录
nim.V2NIMLoginService.login(...)
```
